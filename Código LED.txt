#include <xc.h>
#define _XTAL_FREQ 20000000



void main(void) {
 TRISB = 0b11111111;
 TRISC = 0b00000000;
 
  
 
 while (1) {

        if (PORTBbits.RB2 == 1) {   
            PORTCbits.RC7 = 1;      
            __delay_ms(1000);
        }
        
        if (PORTBbits.RB2 == 0) {   
            PORTCbits.RC7 = 0;      
            __delay_ms(1000);
        }
    }
 
    return;
}
